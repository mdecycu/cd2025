籃球遊戲場景的設置與得分感測

basketball.wbt

#VRML_SIM R2023b utf8

WorldInfo {
  basicTimeStep 32
}

Viewpoint {
  orientation -0.577 0.577 0.577 2.094
  position 0 1 3
}

Floor {
  size 5 5
}

# Circular Loop
Track {
  radius 1
  height 0.1
  thickness 0.1
  color 0.8 0.8 0.8
}

# Balls
DEF BALL1 Solid {
  translation 0.5 0.1 0
  children [
    Shape {
      appearance Appearance {
        material Material {
          diffuseColor 1 0 0
        }
      }
      geometry Sphere {
        radius 0.05
      }
    }
  ]
  physics Physics {
    density 1000
  }
}

DEF BALL2 Solid {
  translation -0.5 0.1 0
  children [
    Shape {
      appearance Appearance {
        material Material {
          diffuseColor 0 0 1
        }
      }
      geometry Sphere {
        radius 0.05
      }
    }
  ]
  physics Physics {
    density 1000
  }
}

# Distance Sensor
DEF DISTANCE_SENSOR DistanceSensor {
  translation 0 0.1 1
  rotation 0 1 0 1.5708
  type "infra-red"
}

controller script:

from controller import Robot

# Time step of the simulation
TIME_STEP = 32

# Create the Robot instance
robot = Robot()

# Get the distance sensor
distance_sensor = robot.getDevice('DISTANCE_SENSOR')
distance_sensor.enable(TIME_STEP)

# Counter for balls
ball_count = 0
threshold = 0.1  # Adjust based on your sensor's distance range

while robot.step(TIME_STEP) != -1:
    # Read the sensor value
    sensor_value = distance_sensor.getValue()
    
    # Check if a ball is detected
    if sensor_value < threshold:
        ball_count += 1
        print(f"Ball count: {ball_count}")
        
        # Wait until the ball passes
        while distance_sensor.getValue() < threshold:
            robot.step(TIME_STEP)

距離感測與接觸感測的差異:

DistanceSensor

  • Functionality: Measures the distance to an object in front of it.
  • Type: Can be an infra-red, laser, ultrasonic, or other types of distance measuring sensors.
  • Range: Can measure distances within a specified range, typically from a few centimeters to several meters.
  • Use Case: Ideal for detecting objects at a distance, obstacle avoidance, measuring distances, and mapping environments.

TouchSensor

  • Functionality: Detects physical contact or touch.
  • Type: Can be a bumper, force sensor, or force3D sensor.
  • Range: Detects contact at a specific point or over an area.
  • Use Case: Ideal for detecting collisions, pressure, and touch. Commonly used for bumpers on robots, detecting when a robot has touched or collided with an object.

Example of Usage:

DistanceSensor

設置:

#VRML_SIM R2023b utf8

WorldInfo {
  basicTimeStep 32
}

Viewpoint {
  position 0 1 3
}

Robot {
  controller "distance_sensor_controller"
  
  DistanceSensor {
    name "distance_sensor"
    type "infra-red"
    translation 0 0.1 0.2
  }
}

控制程式:

from controller import Robot

robot = Robot()
timestep = int(robot.getBasicTimeStep())

distance_sensor = robot.getDevice('distance_sensor')
distance_sensor.enable(timestep)

while robot.step(timestep) != -1:
    distance = distance_sensor.getValue()
    print(f"Distance: {distance}")

最後, 利用 touchsensor 與 display 顯示籃球進框得分:

場景設置:

#VRML_SIM R2023b utf8

WorldInfo {
  basicTimeStep 32
}

Viewpoint {
  orientation -0.577 0.577 0.577 2.094
  position 0 1 3
}

Floor {
  size 5 5
}

# Circular Loop for the basketball hoop
Track {
  radius 0.5
  height 0.05
  thickness 0.05
  color 0.8 0.8 0.8
  translation 0 1 0
  rotation 1 0 0 1.5708
}

# Basketball
DEF BASKETBALL Solid {
  translation 0 1.2 0
  children [
    Shape {
      appearance Appearance {
        material Material {
          diffuseColor 1 0.5 0
        }
      }
      geometry Sphere {
        radius 0.1
      }
    }
  ]
  physics Physics {
    density 1
  }
}

# Touch Sensor to detect the basketball passing through the hoop
DEF TOUCH_SENSOR TouchSensor {
  translation 0 1 0
  rotation 1 0 0 1.5708
  type "force"
}

# Display node to show the score
DEF SCORE_DISPLAY Display {
  width 64
  height 64
  translation 0 2 0
}

控制程式:

from controller import Robot, Display, TouchSensor

# Create the Robot instance
robot = Robot()

# Get the time step of the current world
timestep = int(robot.getBasicTimeStep())

# Get the touch sensor and enable it
touch_sensor = robot.getDevice('TOUCH_SENSOR')
touch_sensor.enable(timestep)

# Get the display node
display = robot.getDevice('SCORE_DISPLAY')

# Initialize the score
score = 0

# Function to update the display with the current score
def update_display(score):
    display.setAlpha(1.0)
    display.setColor(0x000000)
    display.fillRectangle(0, 0, display.getWidth(), display.getHeight())
    display.setAlpha(1.0)
    display.setColor(0xFFFFFF)
    display.setFont("Arial", 24, True)
    display.drawText(f"Score: {score}", 10, 10)

# Main loop
while robot.step(timestep) != -1:
    # Check if the touch sensor is pressed
    if touch_sensor.getValue() > 0:
        score += 1
        update_display(score)
        
        # Wait until the ball passes
        while touch_sensor.getValue() > 0:
            robot.step(timestep)

若需要籃球投出後, 動用 supervisor 控制模式, 在原投球位置產生新球, 則使用下列場景與控制程式:

設置:

#VRML_SIM R2023b utf8

WorldInfo {
  basicTimeStep 32
}

Viewpoint {
  orientation -0.577 0.577 0.577 2.094
  position 0 1 3
}

Floor {
  size 5 5
}

# Circular Loop for the basketball hoop
Track {
  radius 0.5
  height 0.05
  thickness 0.05
  color 0.8 0.8 0.8
  translation 0 1 0
  rotation 1 0 0 1.5708
}

# Initial Basketball
DEF BASKETBALL Solid {
  translation 0 1.2 0
  children [
    Shape {
      appearance Appearance {
        material Material {
          diffuseColor 1 0.5 0
        }
      }
      geometry Sphere {
        radius 0.1
      }
    }
  ]
  physics Physics {
    density 1
  }
}

# Touch Sensor to detect the basketball passing through the hoop
DEF TOUCH_SENSOR TouchSensor {
  translation 0 1 0
  rotation 1 0 0 1.5708
  type "force"
}

# Display node to show the score
DEF SCORE_DISPLAY Display {
  width 64
  height 64
  translation 0 2 0
}

控制程式:

from controller import Supervisor

# Create the Supervisor instance
supervisor = Supervisor()

# Get the time step of the current world
timestep = int(supervisor.getBasicTimeStep())

# Get the touch sensor and enable it
touch_sensor = supervisor.getDevice('TOUCH_SENSOR')
touch_sensor.enable(timestep)

# Get the display node
display = supervisor.getDevice('SCORE_DISPLAY')

# Get the basketball node
basketball = supervisor.getFromDef('BASKETBALL')

# Initialize the score
score = 0

# Function to update the display with the current score
def update_display(score):
    display.setAlpha(1.0)
    display.setColor(0x000000)
    display.fillRectangle(0, 0, display.getWidth(), display.getHeight())
    display.setAlpha(1.0)
    display.setColor(0xFFFFFF)
    display.setFont("Arial", 24, True)
    display.drawText(f"Score: {score}", 10, 10)

# Main loop
while supervisor.step(timestep) != -1:
    # Check if the touch sensor is pressed
    if touch_sensor.getValue() > 0:
        score += 1
        update_display(score)
        
        # Get the position of the current basketball
        ball_position = basketball.getPosition()
        
        # Remove the old basketball
        basketball.remove()

        # Create a new basketball at the position of the old one
        new_ball = supervisor.getFromDef('BASKETBALL')
        new_ball_node = new_ball.clone()
        new_ball_node.getField('translation').setSFVec3f(ball_position)

        # Wait until the ball passes
        while touch_sensor.getValue() > 0:
            supervisor.step(timestep)
